-내 풀이

```java
import java.util.*;
import java.io.*;

class Solution
{
    public static StringBuilder sb= new StringBuilder();
	public static void main(String args[]) throws Exception
	{
		
		Scanner sc = new Scanner(System.in);
		int T;
		T=sc.nextInt();

		for(int test_case = 1; test_case <= T; test_case++)
		{
            long buy=0;long sale=0;long count=0;
			int n = sc.nextInt();
            int[] price = new int[n];
            boolean now = false; // 지금 살건지
            for( int i=0;i<n;i++){
                price[i] += sc.nextInt();
            }
            //작아지는 부분에서 뒤에 더 큰 값이 있다면 구입
            for( int i=1;i<n;i++){
                if(price[i-1]<=price[i]){
                    now = true;
                    count++;
                    if(i==n-1){ //마지막 원소라면 팔기
                        sale+= count*price[i];
                    }
                }else{
                    for(int j=i+1; j<n;j++){
                        if(price[i-1]<price[j])  {
                            now = true;  
                            count++; 
                            break;
                        }
                     	now =false;
                    }
                 }
         
                 if(now){
                     buy += price[i-1];
                     
                  }else{
                 sale += count*price[i-1]; count=0;}
                
                now =false;
            }
            
            sb.append("#"+test_case+" ").append(sale-buy).append("\n");
		}
        System.out.println(sb);
	}
}
```

```java
import java.util.*;
import java.io.*;

class Solution
{
    public static StringBuilder sb= new StringBuilder();
	public static void main(String args[]) throws Exception
	{
		
		Scanner sc = new Scanner(System.in);
		int T;
		T=sc.nextInt();
	
        
		for(int test_case = 1; test_case <= T; test_case++)
		{
            int max = 0;
			int n = sc.nextInt();
            int[] price = new int[n];
 			long remainder = 0; // 차 누적 result
            for( int i=0;i<n;i++){
                price[i] += sc.nextInt();
            }
            
            //뒤에서부터 출발
            for(int i=n-1;i>=0;i--){
            	if(price[i]>max) max = price[i]; //최댓값 갱신
                remainder += max-price[i]; // max와 차이계산
            }
            sb.append("#"+test_case+" ").append(remainder).append("\n");
		}
        System.out.println(sb);
	}
}
```
